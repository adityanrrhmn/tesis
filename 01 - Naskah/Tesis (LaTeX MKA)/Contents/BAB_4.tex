\begin{algorithm}[H]
\caption{Generate Semantic-Based Instructions}
\label{alg:semantic_instruction}
\begin{algorithmic}[1]
\Procedure{GenerateInstructions}{KeyID}
    \State word $\gets$ GetSpecialWord(KeyID)
    \State mapped $\gets$ FindParallelWord(word) \textbf{or} word
    \State embedding $\gets$ GenerateEmbedding(mapped)
    \State nearest $\gets$ FindNearestWords(embedding, n)
    \State instructions $\gets$ [ ]
    \ForAll{w \textbf{in} nearest}
        \State alt $\gets$ FindParallelWord(w) \textbf{or} w
        \State instructions.append(alt)
    \EndFor
    \State \Return CombineIntoInstruction(instructions)
\EndProcedure
\end{algorithmic}
\end{algorithm}
